                Controlling movement in the x-axis
                Left - 0
                Right - 3
                Up - 2
                Down - 1

                from retropy import Rect, screen, actor
                import assets

                player = actor(assets.player, 20, 20, 300, 80, 0, 0, 224, 0, 224)

                def update(dt):
                    pass

                def draw():
                    screen.draw.clear()
                    player.draw()

                def on_key_down(key):
                    if (key == 0): #keys.LEFT
                        player.speed_x(-80)    
                    elif (key == 3): #keys.RIGHT
                        player.speed_x(80)    
        
                def on_key_up(key):
                    pass  

                screen.run(update, draw, on_key_down, on_key_up)
